Arch linux installation guide by Ewancoder
Version: 1.0, 2014

Current PT:
/dev/sda1 - windows system  [48G]
/dev/sda2 - extended        [64G]
    /dev/sda5 - /boot       [128M]
    /dev/sda6 - /           [16G]
    /dev/sda7 - /home       [32G]

1:  Partitioning, formatting, mounting, copy post-manual
        fdisk
        mkfs.ext4
        mount
        cp post* /mnt/home

2:  Forming mirrorlist
        vi /etc/pacman.d/mirrorlist
        pacman -Syy

3:  Installing base-system
        pacstrap -i /mnt base (base-devel)
            i -> avoid auto-confirmation of package selections

4:  Generate fstab
        genfstab    -U -p /mnt >> /mnt/etc/fstab
                    -L
            -p -> avoid printing pseudofs mounts
        vi /mnt/etc/fstab
            add "discard" option for ssd
            comment /boot

5:  Going into new chroot environment
        arch-chroot /mnt /bin/bash

6:  Set timezone
        ln -s /usr/share/zoneinfo/Europe/Minsk /etc/localtime

7:  Set hostname
        echo ewanhost > /etc/hostname

8:  Setup user password
        passwd

9:  Install grub2
        pacman -S grub
        grub-install --target=i386-pc --recheck /dev/sda
            --target=i386-pc - install for BIOS-systems only
                *need only for ensurance, goes for default
        pacman -S os-prober
        grub-mkconfig -o /boot/grub/grub.cfg
            -o -> output file

10: Reboot onto fresh system
        exit
        umount -R /mnt
        rm inst*
        reboot
