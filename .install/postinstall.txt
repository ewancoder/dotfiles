Arch linux post-installation guide by Ewancoder
Version: 1.0, 2014

1:  Configure network
        ping google.com
        ls -la /sys/class/net
        cp /etc/netctl/examples/ethernet-static /etc/netctl/
        vi /etc/netctl/ethernet-static
        [   Interface=enp2s0
            Address='192.168.100.22/24'
            Gateway='192.168.100.1'
            DNS='192.168.100.1' ]
        netctl enable ethernet-static
        netctl start ethernet-static

2:  Add new user
        groupadd fuse
        useradd -m -g users -G fuse -s /bin/bash ewancoder
            -m - create home directory

3:  Install sudo
        pacman -S sudo
        visudo
        passwd ewancoder
        exit
        <login as user>

4:  Install yaourt
        curl -O https://aur.archlinux.org/packages/pa/package-query/package-query.tar.gz <- depencency of yaourt
        tar xvf package-query.tar.gz
        cd package-query
        makepgk -s
            -s -> install all needed dependencies
        #pacman -U package-query.xz
        curl -O https://aur.archlinux.org/packages/ya/yaourt/yaourt.tar.gz
        tar xvf yaourt.tar.gz
        cd yaourt
        makepkg
        #pacman -U yaourt.xz
        cd && rm -r package-query*

5:  Install git and download repos
        yaourt -S git
        git config --global user.name   ewancoder
                            user.email  ewancoder@gmail.com
                            merge.tool  vimdiff
                            core.editor vim
        #ln -s ~/.gitconfig /root/
        git clone https://github.com/ewancoder/dotfiles.git .dotfiles
        #git clone https://github.com/ewancoder/etc.git /etc/.dotfiles
        cd .dotfiles && git submodule update --init --recursive .oh-my-zsh .vim/bundle/vundle
        mkdir .vim/{swap,backup}

6:  Merge ALL GIT files (~/.dotfiles & /etc/.dotfiles) [PREFERABLY DO THIS AFTER INSTALLING ALL SOFTWARE]
        [ .canto, .config, .wmii-hg ] are not links
        [ bitlbee, default, grub.d, pulse, systemd/system, X11/xorg.conf.d ] are not links
        [ .git*, grub.jpg, .install, scripts ] are just files/folders
        #chmod -x /etc/grub.d/10_linux
        #grub-mkconfig -o /boot/grub/grub.cfg <- update background
        #locale-gen <- generate locales
        Setup gvim for root
            #ln -s ~/.vim /root/.vim
            #ln -s ~/.vimrc /root/.vimrc
        irssi make config with password from config_sample
        link all from 'scripts' folder to '/usr/bin' :)
        setfont cyr-sun16 <- set russian font
        yaourt -Syy <- multilib update

7:  Configure audio
        yaourt
            alsa-plugins
            alsa-utils
            pulseaudio
            pulseaudio-alsa
                (lib32-libpulse lib32-alsa-plugins) <- needed by skype
        pulseaudio --start
            pactl list | less
            pacmd set-default-sink 1
            pacmd set-default-source 4

8:  Add ms fonts
        mount windows
        copy windows/msfonts /usr/share/fonts/winfonts
        fc-cache -fv
            -fv - force, verbose

9:  System fixes
        Change 'dead actute' -> 'dead acute' in /usr/share/X11/locale/en_US.UTF-8/Compose
